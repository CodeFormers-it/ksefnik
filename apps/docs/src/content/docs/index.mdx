## ⚡ SDK (TypeScript)

```typescript
import { createKsefnik } from '@ksefnik/core'
import { createKsefSimulator } from '@ksefnik/simulator'

const { adapter } = createKsefSimulator({
  scenario: 'happy-path',
  invoices,
})

const ksef = createKsefnik({
  config: { nip: '5213456784', environment: 'test' },
  adapter,
})

const txs = await ksef.bank.importFromString(csv)
const report = await ksef.reconciliation.run(invoices, txs)
// → matchedCount: 10, confidence: 95%
```
