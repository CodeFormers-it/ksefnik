## Szybki przykład

```typescript
import { createKsefnik } from '@ksefnik/core'
import { createKsefSimulator } from '@ksefnik/simulator'

// Utwórz symulator z gotowym scenariuszem
const { adapter } = createKsefSimulator({ scenario: 'happy-path', invoices })
const ksef = createKsefnik({
  config: { nip: '5213456784', environment: 'test' },
  adapter,
})

// Importuj wyciąg bankowy i uruchom uzgadnianie
const transactions = await ksef.bank.importFromString(csvContent)
const report = await ksef.reconciliation.run(invoices, transactions)

console.log(report.summary)
// { matchedCount: 10, unmatchedInvoices: 2, unmatchedTransactions: 1 }
```
